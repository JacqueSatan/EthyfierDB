1. Download EthyfierDB

Currently only available for Windows

1. 1. Build yourself

use git clone https://github.com/JacqueSatan/EthyfierDB, and open EthyfierDB/EthyfierDB.sln with Visual Studio.

1. 2. Download pre-built

Available at : https://github.com/JacqueSatan/EthyfierDB/releases

1. 3. Include

Add the EthyfierDB directory to the additionnal include directories of your project. To use EthyfierDB, #include "EthyfierDB/ethyfierdb.h"

⚠️ - Only supports std::wstring and wchar_t. You can't use std::string or char with EthyfierDB.

2. How to use EthyfierDB

2. 1. Open a database

#include "EthyfierDB/ethyfierdb.h"
#include <iostream>
#include <string>
int main()
{
// Opens a database at "C:/YourApp/data".
// "C:/YourApp" must exist.
EthyfierDB::DB Database = EthyfierDB::DB(L"C:/YourApp/data");
}

Top

2. 2. Add data to the database

// "db.data()" returns an "EthyfierDB::Object".
// Simple data types
db.data().Int16Items().set(EthyfierDB::Int16(L"Int16 item", 0x1234));
db.data().Int32Items().set(EthyfierDB::Int32(L"int32 Item", 0x12345678));
db.data().StringItems().set(EthyfierDB::String(L"string", L"item value"));
// Complex data types
EthyfierDB::Object object = EthyfierDB::Object(L"object");
object.Int16Items().set(EthyfierDB::Int16(L"another int16", 0x1234));
db.data().ObjectItems().set(object);

2. 3. Replace data in the database

// When an item has the same name, it replaces the old one.
db.data().Int16Items().set(EthyfierDB::Int16(L"Int16 item", 0x1234));
// Items can be modified with a function too.
db.data().Int16Items().get(L"Int16 item").setValue(0xffff);

Top

2. 4. Remove Items from the database

// Items can be deleted by their name.
db.data().Int16Items().remove(L"Int16 item");

2. 5. Get Items from the database

2. 5. 1. Find by name

try
{
// Find a String by name :
std::wstring value = db.data().StringItems().get(L"string").getValue();
value = db.data().StringItems()[L"string"].getValue();
value = db.data()[L"string"].as<EthyfierDB::String>().getValue();
}
// Thrown if the item doesn't exist.
catch (EthyfierDB::Exception& ex)
{
if (ex.exceptionType() == EthyfierDB::ExceptionType::ItemNotFound)
std::wcout << ex.description();
}

2. 5. 2. Find by value

// Returns the item for which the lambda returns true.
db.data().StringItems().find([](EthyfierDB::String& item)
{
return item.getValue() == L"item value";
});
